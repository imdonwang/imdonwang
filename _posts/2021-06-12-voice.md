“不知疲倦地翻越，每一个山丘。”时光飞逝，“华语流行音乐教父”李宗盛翻越人生的山丘和罅隙，从少年意气唱到华发丛生，自省的颗粒烟嗓唱出了人生的低回与高潮，他将 2013 年至 2015 年之间的世界巡回演唱会命名为“既然青春留不住，还是做个大叔好”，颇具自嘲自乐的意味。发行于 2016 年的这张巡演专辑，收录了李宗盛在台北、纽约、伦敦三站演唱会上演出的 29 首精彩曲目，既有他自己的经典作品，也有他为其他华语歌手所写作的曲目。李宗盛用世俗烟火气的质朴唱腔将平凡人生的爱憎和起落娓娓道来，用质朴的原声木吉他、鼓和浪漫的小提琴制造极简的舞台效果，带领听众参与他的人生故事，回溯每个人的青春岁月。

<script>
    $(function() {
        var voiceList = ['0114385747', '0114385748', '0114385750', '0114385753', '0114385754', '0114385755'];
        var flag = 0;
        var index = 0;
        var player = $('#player');
        player.prop("volume", 0.2);
        function mute(f) {
            if(flag === 0) {
                player.attr('src', '/assets/voice_memos/' + voiceList[0] + '.mp3');
                flag++;
            }
            player.prop('muted', f);
            if(f) {
                $('#volume').attr('class', 'fal fa-volume-mute');
                $('.button').css({'color': 'red'});
            } else {
                $('#volume').attr('class', 'fal fa-volume');
                $('.button').css({'color': '#888'});
            }
        } 
        $('#volume').click(function() {
            if(player.prop('muted')) {
                mute(false);
            } else {
                mute(true);
            }
        });
        $('.bg').click(function() {
            mute(false);
            player.attr('src', '/assets/voice_memos/' + voiceList[index] + '.mp3');
            index++;
            if(index >= voiceList.length) {
                index = 0;
            }
        });
    });
</script>
