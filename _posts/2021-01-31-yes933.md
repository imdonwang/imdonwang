<script>
    var endTime = new Date().getTime();
    var playButton = document.getElementById('play_or_pause');
    var script = document.createElement ("script");
    script.type = "text/javascript";
    script.src = "/assets/js/videojs/video.min.js";
    script.onload = function() {
        function loadPlaying() {
            if(endTime < new Date().getTime()) {
                $.get("//api.imdon.wang:8000/yes933", function(res) {
                    endTime = res.end_time;
                    title = res.title + ' - ' + res.artist_name;
                    bgImgURL = "//api.imdon.wang:8000/itunes?term=" + title;
                    if(res.time > res.end_time) {
                        title = "YES 933";
                        bgImgURL = "";
                    }
                    document.title = title;
                    $('#title').html(title);
                    $('#show_title').html(res.show_title);
                    if(bgImgURL) {
                        $.get(bgImgURL, function(res) {
                            if(res.resultCount > 0) {
                                var artUrl = res.results[0].artworkUrl100;
                                artUrl = artUrl.replace('100x100', '300x300');
                                $('.bg').css({"background": "url(" + artUrl + ")", "background-size": "300px"});
                                $('#title').html('<a href="' + res.results[0].trackViewUrl + '" title="去Apple Music查看" style="color: #FA2D48;" target="blank">' + title + '</a>');
                            }
                        }, "json");
                    } else {
                        $('.bg').css({"background": "url(//imdon.wang/assets/img/radio/yes933.png)", "background-size": "300px"});
                    }
                });
            }
        }
        loadPlaying();
        setInterval(function() {
            loadPlaying();
        }, 7000);
        var player = videojs('player', {}, function() {
                this.on("loadstart", function(){});
                this.on("progress", function(){});
                this.on("loadedmetadata", function(){});
                this.on("canplaythrough", function(){
                    playButton.className = "fas fa-pause-circle";
                });
                this.on("waiting", function(){
                    playButton.className = "fas fa-meh-rolling-eyes";
                });
                this.on("play", function(){});
                this.on("playing", function(){
                    playButton.className = "fas fa-pause-circle";
                });
                this.on("pause", function(){
                    playButton.className = "fas fa-play-circle";
                });
                this.on("error", function(){});
                this.on("stalled",function(){});
            }
        );
        playButton.onclick = function() {
            if(player.paused()) {
                player.play();
            } else {
                player.pause();
            }
        }
    };
    document.body.appendChild(script);
</script>
